224 template <typename CollisionObject>
225 bool IsEvtSelected(CollisionObject const& collision, float& centormult)
226 {
227  using namespace filteranalysistask;
228 
229  bool trigsel = false;
230  if (fDataType != kData) {
231  trigsel = true;
232  } else if (collision.alias()[kINT7]) {
233  if (collision.sel7()) {
234  trigsel = true;
235  }
236  }
237 
238  bool zvtxsel = false;
239  /* TODO: vertex quality checks */
240  if (zvtxlow < collision.posZ() and collision.posZ() < zvtxup) {
241  zvtxsel = true;
242  }
243 
244  bool centmultsel = false;
245  switch (fCentMultEstimator) {
246  case kV0M:
247  if (collision.centV0M() < 100 and 0 < collision.centV0M()) {
248  centormult = collision.centV0M();
249  centmultsel = true;
250  }
251  break;
252  default:
253  break;
254  }
255  return trigsel and zvtxsel and centmultsel;
256 }

306 template <typename TrackObject>
307 inline void AcceptTrack(TrackObject const& track, bool& asone, bool& astwo)
308 {
309 
310  asone = false;
311  astwo = false;
312 
313  /* TODO: incorporate a mask in the scanned tracks table for the rejecting track reason */
314  if (matchTrackType(track)) {
315  if (ptlow < track.pt() and track.pt() < ptup and etalow < track.eta() and track.eta() < etaup) {
316  if (((track.sign() > 0) and (trackonecharge > 0)) or ((track.sign() < 0) and (trackonecharge < 0))) {
317  asone = true;
318  }
319  if (((track.sign() > 0) and (tracktwocharge > 0)) or ((track.sign() < 0) and (tracktwocharge < 0))) {
320  astwo = true;
321  }
322  }
323  }

351  template <typename TrackListObject>
352  void filterTracks(TrackListObject const& ftracks)
353  {
354  using namespace filteranalysistask;
355 
356  for (auto& track : ftracks) {
357  /* before track selection */
358  fhPtB->Fill(track.pt());
359  fhEtaB->Fill(track.eta());
360  fhPhiB->Fill(track.phi());
361  fhEtaVsPhiB->Fill(track.phi(), track.eta());
362  fhPtVsEtaB->Fill(track.eta(), track.pt());
363  if (track.sign() > 0) {
364  fhPtPosB->Fill(track.pt());
365  } else {
366  fhPtNegB->Fill(track.pt());
367  }
368 
369  /* track selection */
370  /* tricky because the boolean columns issue */
371  bool asone, astwo;
372  AcceptTrack(track, asone, astwo);
373  if (asone or astwo) {
374  /* the track has been accepted */
375  fhPtA->Fill(track.pt());
376  fhEtaA->Fill(track.eta());
377  fhPhiA->Fill(track.phi());
378  fhEtaVsPhiA->Fill(track.phi(), track.eta());

380  if (track.sign() > 0) {
381  fhPtPosA->Fill(track.pt());
382  } else {
383  fhPtNegA->Fill(track.pt());
384  }
385  }
386  scannedtracks((uint8_t)asone, (uint8_t)astwo);
387  }

587  int etaix = int((t.eta() - etalow) / etabinwidth);
588  /* consider a potential phi origin shift */
589  float phi = GetShiftedPhi(t.phi());
590  int phiix = int((phi - philow) / phibinwidth);
591  return etaix * phibins + phiix;

603  template <typename TrackObject>
604  int GetDEtaDPhiGlobalIndex(TrackObject const& t1, TrackObject const& t2)
605  {
606  using namespace correlationstask;
607 
608  /* rule: ix are always zero based while bins are always one based */
609  int etaix_1 = int((t1.eta() - etalow) / etabinwidth);
610  /* consider a potential phi origin shift */
611  float phi = GetShiftedPhi(t1.phi());
612  int phiix_1 = int((phi - philow) / phibinwidth);
613  int etaix_2 = int((t2.eta() - etalow) / etabinwidth);
614  /* consider a potential phi origin shift */
615  phi = GetShiftedPhi(t2.phi());
616  int phiix_2 = int((phi - philow) / phibinwidth);
617 
618  int deltaeta_ix = etaix_1 - etaix_2 + etabins - 1;
619  int deltaphi_ix = phiix_1 - phiix_2;
620  if (deltaphi_ix < 0) {
621  deltaphi_ix += phibins;
622  }
623 
624  return fhN2_vsDEtaDPhi[kOO]->GetBin(deltaeta_ix + 1, deltaphi_ix + 1);

675  template <typename TrackListObject>
676  void processTrackPairs(TrackListObject const& trks1, TrackListObject const& trks2, int pix, float cmul)
677  {
678  /* process pair magnitudes */
679  double n2 = 0;
680  double sum2PtPt = 0;
681  double sum2DptDpt = 0;
682  double n2nw = 0;
683  double sum2PtPtnw = 0;
684  double sum2DptDptnw = 0;
685  for (auto& track1 : trks1) {
686  double ptavg_1 = 0.0; /* TODO: load ptavg_1 for eta1, phi1 bin */
687  double corr1 = 1.0; /* TODO: track correction weights */
688  for (auto& track2 : trks2) {
689  /* checkiing the same track id condition */
690  if (track1 == track2) {
691  /* exclude autocorrelations */
692  continue;
693  } else {
694  /* process pair magnitudes */
695  double ptavg_2 = 0.0; /* TODO: load ptavg_2 for eta2, phi2 bin */
696  double corr2 = 1.0; /* TODO: track correction weights */
697  double corr = corr1 * corr2;
698  double dptdpt = (track1.pt() - ptavg_1) * (track2.pt() - ptavg_2);
699  n2 += corr;
700  sum2PtPt += track1.pt() * track2.pt() * corr;
701  sum2DptDpt += corr * dptdpt;
702  n2nw += 1;
703  sum2PtPtnw += track1.pt() * track2.pt();
704  sum2DptDptnw += dptdpt;
705  /* get the global bin for filling the differential histograms */
706  int globalbin = GetDEtaDPhiGlobalIndex(track1, track2);
707  fhN2_vsDEtaDPhi[pix]->AddBinContent(globalbin, corr);
708  fhSum2DptDpt_vsDEtaDPhi[pix]->AddBinContent(globalbin, corr * dptdpt);
709  fhSum2PtPt_vsDEtaDPhi[pix]->AddBinContent(globalbin, track1.pt() * track2.pt() * corr);
710  fhN2_vsPtPt[pix]->Fill(track1.pt(), track2.pt(), corr);
711  }
712  }
713  }
714  fhN2_vsC[pix]->Fill(cmul, n2);
715  fhSum2PtPt_vsC[pix]->Fill(cmul, sum2PtPt);
716  fhSum2DptDpt_vsC[pix]->Fill(cmul, sum2DptDpt);
717  fhN2nw_vsC[pix]->Fill(cmul, n2nw);
718  fhSum2PtPtnw_vsC[pix]->Fill(cmul, sum2PtPtnw);
719  fhSum2DptDptnw_vsC[pix]->Fill(cmul, sum2DptDptnw);
720  /* let's also update the number of entries in the differential histograms */
721  fhN2_vsDEtaDPhi[pix]->SetEntries(fhN2_vsDEtaDPhi[pix]->GetEntries() + n2);
722  fhSum2DptDpt_vsDEtaDPhi[pix]->SetEntries(fhSum2DptDpt_vsDEtaDPhi[pix]->GetEntries() + n2);
723  fhSum2PtPt_vsDEtaDPhi[pix]->SetEntries(fhSum2PtPt_vsDEtaDPhi[pix]->GetEntries() + n2);
724  }
